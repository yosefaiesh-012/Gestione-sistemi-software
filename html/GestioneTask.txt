  <script>
    function salvaTask(event) {
      event.preventDefault();
      alert("Task salvato correttamente.");
    }

    function salvaManuale() {
      const titolo = document.getElementById("titolo").value.trim();

      if (titolo === "") {
        alert("Errore: il titolo del task è obbligatorio.");
        return;
      }

      alert("Task salvato correttamente.");
    }

    function assegnaTask() {
      const assegnato = document.getElementById("assegnato").value;
      alert("Task assegnato a: " + assegnato);
    }

    function completaTask() {
      document.getElementById("stato").value = "Completato";
      alert("Stato aggiornato a Completato.");
    }

    function annulla() {
      alert("Operazione annullata.");
    }
  </script>
