  <script>
    function effettuaLogin(event) {
      event.preventDefault();

      const username = document.getElementById("username").value.trim();
      const password = document.getElementById("password").value.trim();
      const ruolo = document.getElementById("ruolo").value;
      const message = document.getElementById("message");

      if (username === "" || password === "" || ruolo === "") {
        message.className = "message error";
        message.textContent = "Errore: inserire username, password e ruolo.";
        return;
      }

      message.className = "message success";

      if (ruolo === "amministratore") {
        message.textContent = "Accesso effettuato. Reindirizzamento alla home Amministratore.";
      } else if (ruolo === "project-manager") {
        message.textContent = "Accesso effettuato. Reindirizzamento alla home Project Manager.";
      } else if (ruolo === "sviluppatore") {
        message.textContent = "Accesso effettuato. Reindirizzamento alla home Sviluppatore.";
      } else if (ruolo === "tester") {
        message.textContent = "Accesso effettuato. Reindirizzamento alla home Tester.";
      }
    }

    function mostraPassword() {
      const password = document.getElementById("password");
      const button = document.querySelector(".toggle-password");

      if (password.type === "password") {
        password.type = "text";
        button.textContent = "Nascondi";
      } else {
        password.type = "password";
        button.textContent = "Mostra";
      }
    }

    function recuperoPassword() {
      alert("Funzionalità prevista come possibile sviluppo futuro.");
    }
