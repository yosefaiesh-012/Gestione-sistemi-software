  <script>
    function salvaBug(event) {
      event.preventDefault();
      alert("Bug salvato correttamente.");
    }

    function salvaBugManuale() {
      const titolo = document.getElementById("titolo").value.trim();

      if (titolo === "") {
        alert("Errore: il titolo del bug è obbligatorio.");
        return;
      }

      alert("Bug salvato correttamente.");
    }

    function verificaBug() {
      document.getElementById("statoBug").value = "Chiuso";
      alert("Bug verificato con esito positivo. Stato aggiornato a Chiuso.");
    }

    function validaTask() {
      alert("Task validato correttamente. Stato aggiornato a Validato.");
    }

    function riapriElemento() {
      document.getElementById("statoBug").value = "Aperto";
      alert("Elemento riaperto per esito negativo della verifica.");
    }

    function annulla() {
      alert("Operazione annullata.");
    }
